exports.handler = async (event) => { if (event.httpMethod !== 'POST') { return { statusCode: 405, body: JSON.stringify({ error: 'Method Not Allowed' }) }; } try { const { to, name, dueDate } = JSON.parse(event.body); if (!to || !name || !dueDate) { return { statusCode: 400, body: JSON.stringify({ error: 'جميع الحقول مطلوبة' }) }; } const accountSid = process.env.TWILIO_ACCOUNT_SID; const authToken = process.env.TWILIO_AUTH_TOKEN; const fromNumber = process.env.TWILIO_PHONE_NUMBER; // بناء رابط API Twilio const url = `https://api.twilio.com/2010-04-01/Accounts/${accountSid}/Messages.json`; // إرسال الطلب باستخدام fetch المدمج const response = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded', 'Authorization': 'Basic ' + Buffer.from(accountSid + ':' + authToken).toString('base64') }, body: new URLSearchParams({ From: `whatsapp:${fromNumber}`, To: `whatsapp:${to}`, Body: `السلام عليكم ${name}،\nهذا تذكير من وكّل بدفع الإيجار المستحق بتاريخ ${dueDate}.\nنشكر تعاونك.` }).toString() }); const data = await response.json(); if (response.ok) { return { statusCode: 200, body: JSON.stringify({ success: true, sid: data.sid }) }; } else { return { statusCode: 500, body: JSON.stringify({ error: data.message || 'فشل الإرسال' }) }; } } catch (error) { return { statusCode: 500, body: JSON.stringify({ error: error.message }) }; } };
